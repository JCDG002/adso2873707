CREATE DATABASE articulos_db;
USE articulos_db;

CREATE TABLE usuario(
idUsuario INT(10) PRIMARY KEY AUTO_INCREMENT,
nombre VARCHAR(50) NOT NULL,
telefono VARCHAR(20),
password VARCHAR(20) NOT NULL,
genero CHAR(1) NOT NULL
);

CREATE TABLE categoria(
idCategoria SMALLINT PRIMARY KEY AUTO_INCREMENT,
nombre VARCHAR(50) NOT NULL,
icono VARCHAR(255) NOT NULL
);

CREATE TABLE autor(
idAutor SMALLINT PRIMARY KEY AUTO_INCREMENT,
nombre VARCHAR(50) NOT NULL,
fecha_nac DATE NOT NULL,
direccion VARCHAR(30) NOT NULL,
email VARCHAR(255) NOT NULL,
password VARCHAR(255) NOT NULL,
foto_perfil VARCHAR(255) NOT NULL
);

CREATE TABLE articulo(
idArticulo MEDIUMINT PRIMARY KEY AUTO_INCREMENT,
titulo VARCHAR NOT NULL,
descripcion TEXT NOT NULL,
image VARCHAR NOT NULL,
fecha_crea DATETIME NOT NULL,
fkIdCategoria SMALLINT NOT NULL,
fkIdAutor SMALLINT NOT NULL,
FOREIGN KEY (fk
);
